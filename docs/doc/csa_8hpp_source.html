185  inline int minimize(
186  int n,
187  Scalar_x* x,
188  Scalar_fx (*fx)(void*, Scalar_x*),
189  void (*step)(void*, Scalar_x* y, const Scalar_x*, float tgen),
190  void (*progress)(void*,
191  Scalar_fx cost,
192  float tgen,
193  float tacc,
194  int opt_id,
195  int iter),
196  void* instance)
197  {
198  Scalar_fx fx0 = fx(instance, x);
199 
200  // Initialize shared values.
201  SharedStates<Scalar_x, Scalar_fx> shared_states(this->m, n, x, fx0);
202  float tacc = this->tacc_initial;
203  float tgen = this->tgen_initial;
204  float tmp, sum_a, prob_var, gamma = m;
205 
206  omp_lock_t lock;
207  omp_init_lock(&lock);
208 
209  #pragma omp parallel shared(n, shared_states, tacc, tgen, gamma) num_threads(this->m) default(none)
210  {
211  int k, opt_id = omp_get_thread_num();
212 
213  Scalar_fx max_cost = shared_states[0].cost;
214  Scalar_fx cost;
215  std::vector<Scalar_x> y(n, Scalar_x(0));
216  float unif, prob;
217 
218  #pragma omp for
219  for (int iter = 0; iter < this->max_iterations; ++iter) {
220  // Generate a new solution.
221  step(instance, y.data(), shared_states[opt_id].x.data(), tgen);
222  cost = fx(instance, y.data());
223 
224  // Take step and update best solution so far.
225  if (cost < shared_states[opt_id].cost) {
226  omp_set_lock(&lock);
227  // Update best solution for this thread.
228  if (cost < shared_states[opt_id].best_cost) {
229  shared_states[opt_id].best_cost = cost;
230  shared_states[opt_id].best_x = y;
231  if (progress != nullptr)
232  progress(instance, cost, tgen, tacc, opt_id, iter);
233  }
234 
235  shared_states[opt_id].step(y, cost);
236  omp_unset_lock(&lock);
237  } else {
238  unif = drand48();
239  prob = std::exp((shared_states[opt_id].cost - max_cost) / tacc) / gamma;
240 
241  if (prob > unif) {
242  omp_set_lock(&lock);
243  shared_states[opt_id].step(y, cost);
244  omp_unset_lock(&lock);
245  }
246  }
247 
248  // Update temperatures.
249  if (omp_test_lock(&lock)) {
250 
251  // Get maximum cost.
252  max_cost = shared_states[0].cost;
253  for (k = 0; k < this->m; ++k)
254  if (shared_states[k].cost > max_cost)
255  max_cost = shared_states[k].cost;
256 
257  // Update gamma and prob_var;
258  gamma = sum_a = 0.;
259  for (k = 0; k < this->m; ++k) {
260  tmp = (shared_states[k].cost - max_cost) / tacc;
261  gamma += std::exp(tmp);
262  sum_a += std::exp(2.0 * tmp);
263  }
264  prob_var = (this->m * (sum_a / (gamma * gamma)) - 1.) /
265  (this->m * this->m);
266 
267  // Update the acceptance temperature.
268  if (prob_var > this->desired_variance)
269  tacc += this->tacc_schedule * tacc;
270  else
271  tacc -= this->tacc_schedule * tacc;
272 
273  // Update the generation temperature.
274  tgen = this->tgen_schedule * tgen;
275 
276  // Unset the lock.
277  omp_unset_lock(&lock);
278  }
279  }
280  } // End of #pragma omp parallel
281 
282  // Get best result.
283  int best_ind = 0;
284  Scalar_fx best_cost = shared_states[0].best_cost;
285  for (int k = 0; k < this->m; ++k) {
286  if (shared_states[k].best_cost < best_cost) {
287  best_cost = shared_states[k].best_cost;
288  best_ind = k;
289  }
290  }
291  State<Scalar_x,Scalar_fx> best_state = shared_states[best_ind];
292  for (int i = 0; i < n; ++i)
293  x[i] = best_state.best_x[i];
294 
295  // Clean up.
296  omp_destroy_lock(&lock);
297 
298  return 0;
299  }
300 };
